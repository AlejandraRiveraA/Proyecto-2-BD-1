USE `project2`;
DROP procedure IF EXISTS `insert_team`;

DELIMITER $$
USE `project2`$$
CREATE PROCEDURE `insert_team` (IN pcountry int, IN palignment int, IN pcaptain int, IN pname VARCHAR(50))
BEGIN
	
    INSERT INTO team(id_country, id_alignment, id_captain, team_name)
	VALUES(pcountry, palignment, pacaptain, pname);
	commit;

END$$

DELIMITER ;
