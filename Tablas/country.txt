CREATE TABLE `project2`.`country` (
  `id_country` INT NOT NULL COMMENT 'Unique identifier of the country.',
  `country_name` VARCHAR(45) NOT NULL COMMENT 'Name of the country. ',
  `id_continent` INT NOT NULL COMMENT 'Unique identifier of the continent. ',
  PRIMARY KEY (`id_country`),
  UNIQUE INDEX `id_country_UNIQUE` (`id_country` ASC) VISIBLE,
  UNIQUE INDEX `country_name_UNIQUE` (`country_name` ASC) VISIBLE,
  INDEX `id_continent_fk_idx` (`id_continent` ASC) VISIBLE,
  UNIQUE INDEX `id_continent_UNIQUE` (`id_continent` ASC) VISIBLE,
  CONSTRAINT `id_continent_fk`
    FOREIGN KEY (`id_continent`)
    REFERENCES `project2`.`continent` (`id_continent`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
COMMENT = 'Catalogue table that stores all countries.';
